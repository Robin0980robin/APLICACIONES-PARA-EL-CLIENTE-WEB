    <script type="text/javascript">
        var numero1 = parseFloat(prompt("Introduce el primer número:"));
        var numero2 = parseFloat(prompt("Introduce el segundo número:"));

        for (var i = 1; i <= 5; i++) {
            switch (i) {
                case 1:
                    alert("Iteración " + i + ": Suma de los dos números: " + (numero1 + numero2));
                    break;
                case 2:
                    alert("Iteración " + i + ": Resta de los dos números: " + (numero1 - numero2));
                    break;
                case 3:
                    alert("Iteración " + i + ": Multiplicación de los dos números: " + (numero1 * numero2));
                    break;
                case 4:
                    if (numero2 !== 0) {
                        alert("Iteración " + i + ": División de los dos números: " + (numero1 / numero2));
                    } else {
                        alert("Iteración " + i + ": No se puede dividir por cero.");
                    }
                    break;
                case 5:
                    if (numero2 !== 0) {
                        alert("Iteración " + i + ": Módulo de los dos números: " + (numero1 % numero2));
                    } else {
                        alert("Iteración " + i + ": No se puede calcular el módulo con cero.");
                    }
                    break;
            }
        }
    </script>
